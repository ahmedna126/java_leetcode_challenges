package org.example;

public class Main {

    public static String toLowerCase(String s) {
        StringBuilder result = new StringBuilder();
            for(char c : s.toCharArray()){
                if (c >= 'A' && c <= 'Z' ){
                    c = (char) ((int) c + 32 );
                }
            result.append(c);
            }

        return result.toString();
    }

    public static void main(String[] args)
    {

        System.out.println(toLowerCase("Hello"));

    }
}
