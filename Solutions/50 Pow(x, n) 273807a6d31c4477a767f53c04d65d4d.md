package org.example;

public class Main {

    public static double myPow(double x, int n) {
        if (n == 0)
            return 1.0;

        if (n == Integer.MIN_VALUE)
        {
            x = x * x;
            n = n / 2;
        }

        if (n < 0)
        {
            x = 1 / x;
            n = -n;
        }

        double result = 1.0;

        while (n > 0)
        {
            if (n % 2 == 1)
                result *= x;

            x *= x;
            n /= 2;
        }

        return result;
    }
    public static void main(String[] args) {

        System.out.println(myPow(1.00000, 2147483647));

    }
}
